\newcommand{\cbpEmptyPageIfTwoside}[1][empty]{
	\IfStrEqCase{\cbpDocumentSides} {
		{oneside} {}
		{twoside} {
			\newpage
			\IfStrEqCase{#1} {
				{empty} {
					\thispagestyle{empty}
				}
				{numbered} {}
			}
			\mbox{}
			\clearpage
		}
	}
}

\newcommand{\cbpEmptyPageIfTwosideOdd}{
	\checkoddpage
	\ifoddpage
		\cbpEmptyPageIfTwoside[numbered]
	\else
	\fi
}


\newcommand{\cbpEmptyPageIfTwosideEven}{
	\checkoddpage
	\ifoddpage
	\else
		\cbpEmptyPageIfTwoside[numbered]
	\fi
}
