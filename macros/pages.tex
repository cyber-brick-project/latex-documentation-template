\newcommand{\cbpEmptyPageIfTwoside}{
	\IfStrEqCase{\cbpDocumentSides} {
		{oneside} {}
		{twoside} {
			\newpage
			\thispagestyle{empty}
			\mbox{}
			\clearpage
		}
	}
}

\newcommand{\cbpEmptyPageIfTwosideOdd}{
	\checkoddpage
	\ifoddpage
		\cbpEmptyPageIfTwoside
	\else
	\fi
}


\newcommand{\cbpEmptyPageIfTwosideEven}{
	\checkoddpage
	\ifoddpage
	\else
		\cbpEmptyPageIfTwoside
	\fi
}
