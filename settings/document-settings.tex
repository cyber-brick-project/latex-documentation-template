% Document Class
\ifdefined\cbpDocumentClass
	% recommended values:
	% - mwart
	% - report
\else
	\newcommand{\cbpDocumentClass}{mwart}
\fi

% Document sides (book or digital)
\ifdefined\cbpDocumentSides
	% recommended values:
	% - oneside
	% - twoside
\else
	\newcommand{\cbpDocumentSides}{oneside}
\fi

\newcommand{\cbpRunDocumnetClass}{
	\documentclass[12pt,\cbpDocumentSides,a4paper]{\cbpDocumentClass}
}

\newcommand{\cbpRunDocumnetMargins}{
	\IfStrEqCase{\cbpDocumentSides} {
		{oneside} {
			\usepackage[top=2.5cm, bottom=2.5cm, left=2.5cm, right=2.5cm]{geometry}
		}
		{twoside} {
			\usepackage[top=2.5cm, bottom=2.5cm, left=3cm, right=2.5cm]{geometry}
		}
	}
}
